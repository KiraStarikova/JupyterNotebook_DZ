from sympy import *

from sympy.plotting import plot

x=symbols('x')

f=-12*x**4*sin(cos(x))-18*x**3+5*x**2+10*x-30

print(f)

plot(f)

print('зададим интервал от -250 до 250')

plot(f,(x,-250,250), line_color='red')

df=diff(f)

plot(df,(x,-250,250), line_color='green')
